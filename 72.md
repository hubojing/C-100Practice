题目：创建一个链表。
```c
#include <stdio.h>
#include <malloc.h>

struct LNode
{
	int data;
	struct LNode *next;
};

LNode* createList(int n)
{
	LNode *list, *p, *q;
	list = (LNode*)malloc(sizeof(LNode));
	list->next = nullptr;
	q = list;
	for (int i = 0; i <= n; ++i)
	{
		p = (LNode*)malloc(sizeof(LNode));
		printf("请输入第%d个元素的值：", i);
		scanf("%d", &(p->data));
		p->next = nullptr;
		q->next = p;
		q = p;
	}
	return list;
}

void print(LNode* list)
{
	LNode* p = list->next;
	while (p != nullptr)
	{
		printf("%d\n", p->data);
		p = p->next;
	}
}

int main()
{
	LNode* list = nullptr;
	int n;
	scanf("%d", &n);
	list = createList(n);
	printf("链表各值为:\n");
	print(list);

	return 0;
}
```
