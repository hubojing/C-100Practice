题目：从键盘输入一些字符，逐个把它们送到磁盘上去，直到输入一个#为止。
```c
#include<stdio.h>
#include<stdlib.h>

int main()
{
	FILE*fp = NULL;
	char filename[25];
	char ch;
	printf("输入文件名称：\n");
	scanf("%s", filename);
	if ((fp = fopen(filename, "w")) == NULL)
	{
		printf("error: cannot open file!\n");
		exit(0);
	}
	printf("输入字符，以#结束：\n");
	getchar();
	while ((ch = getchar()) != '#') {
		fputc(ch, fp);
	}
	fclose(fp);

	return 0;
}
```
结果：
```
输入文件名称：
test.txt
输入字符，以#结束：
hubojing.github.io
#
```
