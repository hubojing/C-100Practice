题目：输入两个正整数m和n，求其最大公约数和最小公倍数。
最小公倍数 * 最大公约数 = m * n
```c
#include<stdio.h>

int main()
{
	int m, n;
	scanf("%d %d", &m, &n);
	int tmp = m < n ? m : n;
	for (int i = tmp; i > 0; --i)
	{
		if (m%i == 0 && n%i == 0)
		{
			printf("最大公约数为%d\n", i);
			printf("最小公倍数为%d", m*n / i);
			break;
		}
	}


	return 0;
}
```
